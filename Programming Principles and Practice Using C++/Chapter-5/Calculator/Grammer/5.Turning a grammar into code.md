# 🟢 2. Core Structure (VERY IMPORTANT)

👉 File: `parser_demo.cpp`

```cpp
#include <iostream>
#include <vector>

// ---------------- Token ----------------
class Token {
public:
    char kind;
    double value;

    Token(char k) : kind{k}, value{0.0} {}
    Token(char k, double v) : kind{k}, value{v} {}
};

// ---------------- Token storage ----------------
std::vector<Token> tok;
int pos = 0;

// get current token
Token get() {
    return tok[pos++];
}

// put back
void putback() {
    --pos;
}
```

# 🟢 3. Primary (Lowest level)

```cpp
double primary()
{
    Token t = get();

    switch (t.kind) {
    case '8':   // number
        return t.value;

    case '(':
    {
        double d = expression();
        Token t2 = get();
        if (t2.kind != ')') {
            std::cerr << "Error: ')' expected\n";
        }
        return d;
    }

    default:
        std::cerr << "Error: primary expected\n";
        return 0;
    }
}
```

# 🟢 4. Term (handles * /)

```cpp
double term()
{
    double left = primary();

    while (true) {
        Token t = get();

        switch (t.kind) {
        case '*':
            left *= primary();
            break;

        case '/':
            left /= primary();
            break;

        default:
            putback();
            return left;
        }
    }
}
```

# 🟢 5. Expression (handles + -)

```cpp
double expression()
{
    double left = term();

    while (true) {
        Token t = get();

        switch (t.kind) {
        case '+':
            left += term();
            break;

        case '-':
            left -= term();
            break;

        default:
            putback();
            return left;
        }
    }
}
```

👉 `expression()`:

add / subtract handle பண்ணும்
